\newcommand{\element}[7]{
		\psset{fillstyle=solid,framesize=0.95 0.95,framearc=0.1,linestyle=none}
		\fnode[fillcolor=#7](#1,#2){#3}
		\pnode[0,-0.4](#1,#2){name}
		\uput[90](name){\parbox{38mm}{\centering
			{\LARGE\bfseries{}#3}\\
			\footnotesize{}#4\\
			\footnotesize{}#6}}
		\pnode[-0.4,0.25](#1,#2){name} \uput[90](name){\parbox{0mm}{\footnotesize{}#5}}}
%
% The \goup command is used to add the hatch mask for the lanthanoids and actinoids

	\fnode[fillstyle=vlines,hatchcolor=white,hatchsep=5pt,hatchwidth=4pt](#1,#2){#3}
}
